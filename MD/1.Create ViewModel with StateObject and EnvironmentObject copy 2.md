### 🔷 1.Create ViewModel with StateObject and EnvironmentObject

```swift

// in MapViewModel

class MapViewModel: ObservableObject {

	@Published var locations: [Locations] = Bundle.main.decode("awesomeDB.json")
}

```

```swift

//  awesomeKRApp.swift

import SwiftUI

@main
struct awesomeKRApp: App {

	@StateObject private var vm: MapViewModel = MapViewModel()

	var body: some Scene {
		WindowGroup {
			SplashScreen()
				.environmentObject(vm)
		}
	}
}

```

```swift
//  MapView.swift

import SwiftUI

struct MapView: View {

	@EnvironmentObject private var vm: MapViewModel

	var body: some View {
		List {
			ForEach(vm.locations) {
				Text($0.name)
			}
		}
	}
}

struct MapView_Previews: PreviewProvider {
	static var previews: some View {
		MapView()
			.environmentObject(MapViewModel())
	}
}
```
