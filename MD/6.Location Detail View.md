### 🔷 6.Location Detail View

```swift
//  DetailView.swift

import SwiftUI
import MapKit

struct DetailView: View {
// MARK: -  PROPERTY
@EnvironmentObject private var vm: MapViewModel
let location: Location
let accentColor = Color("AccentColor")


// MARK: -  BODY
var body: some View {
ScrollView {
VStack {
  imageSection

  VStack (alignment: .center, spacing: 16) {
    titleSection
    headlineSection
    descriptionSection
    youtubeSection
    dontMissSection
    transportationSection
    mapLayer
    moreInformation
    tourInformation
      .padding(.top, 20)
    copyRight

  } //: VSTACK
  .frame(maxWidth: .infinity, alignment: .leading)
  .padding()
} //: VSTACK
} //: SCROLL
.ignoresSafeArea()
.background(.ultraThinMaterial)
// backBtn
.overlay(alignment: .topLeading) {
backBtn
}
}
}

// MARK: -  PREVIEW
struct DetailView_Previews: PreviewProvider {
static let locations: [Location] = Bundle.main.decode("awesomeDB.json")

static var previews: some View {
DetailView(location: locations[0])
  .environmentObject(MapViewModel())
}
}


// MARK: -  EXTENSION
extension DetailView {

// BACK BTN
private var backBtn: some View {
Button {
  vm.sheetLocation = nil
} label: {
  Image(systemName: "xmark")
    .font(.headline)
    .padding()
    .foregroundColor(accentColor)
    .background(.thickMaterial)
    .cornerRadius(10)
    .shadow(radius: 4)
    .padding()
}

}

// IMAGE SECTION
private var imageSection: some View {
TabView {
  ForEach(location.imageNames, id: \.self) {
    Image($0)
      .resizable()
      .scaledToFill()
  } //: LOOP
} //: TAB
.frame(height: UIScreen.main.bounds.height * 0.35)
.padding(.bottom, 40)
.tabViewStyle(PageTabViewStyle())
.shadow(color: Color.black.opacity(0.3), radius: 20, x: 0, y: 10)
}

// TITLE SECTION
private var titleSection: some View {
VStack(alignment: .center, spacing: 8) {
  Text(location.name)
    .font(.largeTitle)
    .fontWeight(.semibold)
    .padding(.top, 10)

  Text(location.cityName)
    .font(.title3)
    .foregroundColor(.secondary)
} //: VSTACK
}

// HEADLINE SECTION
private var headlineSection: some View {
Text(location.headline)
  .font(.title3)
  .multilineTextAlignment(.center)
  .foregroundColor(.accentColor)
}


// DESCRIPTION SECTION
private var descriptionSection: some View {
VStack(alignment: .leading, spacing: 16) {
  Text(location.description)
    .font(.subheadline)
    .foregroundColor(.secondary)
} //: VSTACK
}

// YOUTUBE SECTION
private var youtubeSection: some View {
YoutubeView(videoID: location.youtubeID)
  .frame(minHeight: 0, maxHeight: UIScreen.main.bounds.height * 0.3)
  .cornerRadius(12)
}

// DONT MISS SECTION
private var dontMissSection: some View {
Group {
  HeadingView(headingImage: "exclamationmark.circle", headingText: "Don't Miss")

  InsetDontMissView(location: location)
}
}

// TRANSPORTATION SECTION
private var transportationSection: some View {
Group {
  HeadingView(headingImage: "bus.fill", headingText: "Transportation")
  Text(location.transportation)
  Text(location.parking)
}
}

// MAP SECTION
private var mapLayer: some View {
Map(coordinateRegion: .constant(MKCoordinateRegion(
  center: location.coordinates,
  span: MKCoordinateSpan(latitudeDelta: 0.01, longitudeDelta: 0.01))),
    annotationItems: [location]) { location in
  MapAnnotation(coordinate: location.coordinates) {
    MapAnnotationPin()
      .shadow(radius: 10)
  }
} //: MAP
    .allowsHitTesting(false) // disabling gestures on the map
    .aspectRatio(1, contentMode: .fill)
    .cornerRadius(10)
}

// More Information
private var moreInformation: some View {
Group {
  HeadingView(headingImage: "info.circle", headingText: "More Infomation")

  Text(location.fee)
  Text(location.availableHours)
  Text(location.others)
}
}

// Tourist Information
private var tourInformation: some View {
GroupBox(
  label: SettingLabelView(
    labelText: "Tourist Information",
    labelImage: "figure.walk")
){
  SettingsRowView(name: "🏠", content: location.address)
  SettingsRowView(name: "📞", content: location.phone)
  SettingsRowView(name: "📲", linkLabel: "\(location.name) Site", linkDestination: location.link)
  SettingsRowView(name: "📂", linkLabel: "Go to Reference", linkDestination: "http://english.visitkorea.or.kr/enu/TRV/TV_ENG_Must100.jsp#")
}
}

// Copyright
private var copyRight: some View {
Text(location.copyright)
  .font(.footnote)
  .foregroundColor(.secondary)
  .multilineTextAlignment(.center)
  .padding()
  .padding(.bottom, 30)
}
}
```
